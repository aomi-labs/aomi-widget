# Custom Chat UI

This example shows how to build a fully custom chat interface using only `@aomi-labs/react` hooks. No widget components are used -- every element is built from scratch with your own design system.

## Scenario

A company wants an AI assistant integrated into their existing dashboard. They have their own design system using Tailwind CSS and do not want the standard Aomi widget UI. They need:

- A message list with their own bubble styles
- A custom input composer
- A thread switcher in their existing sidebar
- Full control over layout and interactions

## Setup

Install the headless library:

```bash
npm install @aomi-labs/react @assistant-ui/react
```

Set up the provider in your app layout:

```tsx
// app/providers.tsx
"use client";

import { AomiRuntimeProvider } from "@aomi-labs/react";

export function Providers({ children }: { children: React.ReactNode }) {
  return (
    <AomiRuntimeProvider backendUrl={process.env.NEXT_PUBLIC_BACKEND_URL!}>
      {children}
    </AomiRuntimeProvider>
  );
}
```

```tsx
// app/layout.tsx
import { Providers } from "./providers";

export default function RootLayout({ children }: { children: React.ReactNode }) {
  return (
    <html lang="en">
      <body>
        <Providers>{children}</Providers>
      </body>
    </html>
  );
}
```

## MessageBubble Component

A styled message bubble that matches the company's design system:

```tsx
// components/chat/message-bubble.tsx
"use client";

type MessageBubbleProps = {
  role: "user" | "assistant";
  children: React.ReactNode;
};

export function MessageBubble({ role, children }: MessageBubbleProps) {
  const isUser = role === "user";

  return (
    <div className={`flex ${isUser ? "justify-end" : "justify-start"} mb-3`}>
      {!isUser && (
        <div className="mr-2 mt-1 h-8 w-8 rounded-full bg-indigo-100 flex items-center justify-center">
          <span className="text-indigo-600 text-xs font-bold">AI</span>
        </div>
      )}
      <div
        className={`max-w-[70%] rounded-2xl px-4 py-3 text-sm leading-relaxed ${
          isUser
            ? "bg-indigo-600 text-white rounded-br-md"
            : "bg-slate-100 text-slate-800 rounded-bl-md"
        }`}
      >
        {children}
      </div>
    </div>
  );
}
```

## MessageList Component

Renders messages from the current thread:

```tsx
// components/chat/message-list.tsx
"use client";

import { useRef, useEffect } from "react";
import { useAomiRuntime } from "@aomi-labs/react";
import { MessageBubble } from "./message-bubble";

export function MessageList() {
  const { currentThreadId, getMessages, isRunning } = useAomiRuntime();
  const messages = getMessages(currentThreadId);
  const bottomRef = useRef<HTMLDivElement>(null);

  // Auto-scroll to bottom on new messages
  useEffect(() => {
    bottomRef.current?.scrollIntoView({ behavior: "smooth" });
  }, [messages, isRunning]);

  return (
    <div className="flex-1 overflow-y-auto px-6 py-4">
      {messages.length === 0 && (
        <div className="flex h-full items-center justify-center text-slate-400">
          Start a conversation...
        </div>
      )}

      {messages.map((msg, i) => (
        <MessageBubble key={i} role={msg.role as "user" | "assistant"}>
          {msg.content.map((block, j) => (
            <span key={j}>{block.type === "text" ? block.text : null}</span>
          ))}
        </MessageBubble>
      ))}

      {isRunning && (
        <MessageBubble role="assistant">
          <span className="inline-flex gap-1">
            <span className="animate-bounce">.</span>
            <span className="animate-bounce" style={{ animationDelay: "0.1s" }}>.</span>
            <span className="animate-bounce" style={{ animationDelay: "0.2s" }}>.</span>
          </span>
        </MessageBubble>
      )}

      <div ref={bottomRef} />
    </div>
  );
}
```

## ChatInput Component

Custom composer with the company's input style:

```tsx
// components/chat/chat-input.tsx
"use client";

import { useState, useRef, useEffect } from "react";
import { useAomiRuntime } from "@aomi-labs/react";

export function ChatInput() {
  const { sendMessage, isRunning, cancelGeneration } = useAomiRuntime();
  const [input, setInput] = useState("");
  const inputRef = useRef<HTMLTextAreaElement>(null);

  // Auto-resize textarea
  useEffect(() => {
    if (inputRef.current) {
      inputRef.current.style.height = "auto";
      inputRef.current.style.height = `${inputRef.current.scrollHeight}px`;
    }
  }, [input]);

  const handleSubmit = async () => {
    if (!input.trim() || isRunning) return;
    const text = input;
    setInput("");
    await sendMessage(text);
  };

  const handleKeyDown = (e: React.KeyboardEvent) => {
    if (e.key === "Enter" && !e.shiftKey) {
      e.preventDefault();
      handleSubmit();
    }
  };

  return (
    <div className="border-t border-slate-200 bg-white px-6 py-4">
      <div className="flex items-end gap-3">
        <textarea
          ref={inputRef}
          value={input}
          onChange={(e) => setInput(e.target.value)}
          onKeyDown={handleKeyDown}
          placeholder="Ask anything..."
          rows={1}
          className="flex-1 resize-none rounded-xl border border-slate-200 px-4 py-3 text-sm
                     focus:border-indigo-300 focus:outline-none focus:ring-2 focus:ring-indigo-100"
          disabled={isRunning}
        />
        {isRunning ? (
          <button
            onClick={cancelGeneration}
            className="rounded-xl bg-red-500 px-5 py-3 text-sm font-medium text-white
                       hover:bg-red-600 transition-colors"
          >
            Stop
          </button>
        ) : (
          <button
            onClick={handleSubmit}
            disabled={!input.trim()}
            className="rounded-xl bg-indigo-600 px-5 py-3 text-sm font-medium text-white
                       hover:bg-indigo-700 disabled:opacity-40 transition-colors"
          >
            Send
          </button>
        )}
      </div>
    </div>
  );
}
```

## ThreadSwitcher Component

Fits into the company's existing sidebar:

```tsx
// components/chat/thread-switcher.tsx
"use client";

import { useAomiRuntime } from "@aomi-labs/react";

export function ThreadSwitcher() {
  const {
    currentThreadId,
    threadMetadata,
    selectThread,
    createThread,
    deleteThread,
  } = useAomiRuntime();

  const threads = Array.from(threadMetadata.entries())
    .sort(([, a], [, b]) => (b.lastActiveAt ?? 0) - (a.lastActiveAt ?? 0));

  return (
    <div className="flex flex-col gap-1">
      <button
        onClick={() => createThread()}
        className="mb-2 w-full rounded-lg border border-dashed border-slate-300 px-3 py-2
                   text-sm text-slate-500 hover:border-indigo-300 hover:text-indigo-600
                   transition-colors"
      >
        + New conversation
      </button>

      {threads.map(([id, meta]) => (
        <button
          key={id}
          onClick={() => selectThread(id)}
          className={`flex items-center justify-between rounded-lg px-3 py-2 text-left text-sm
                      transition-colors ${
            id === currentThreadId
              ? "bg-indigo-50 text-indigo-700 font-medium"
              : "text-slate-600 hover:bg-slate-50"
          }`}
        >
          <span className="truncate">{meta.title || "Untitled"}</span>
          <span
            onClick={(e) => {
              e.stopPropagation();
              deleteThread(id);
            }}
            className="ml-2 text-xs text-slate-400 hover:text-red-500 cursor-pointer"
          >
            x
          </span>
        </button>
      ))}
    </div>
  );
}
```

## Full Page Composition

Assemble all components on the chat page:

```tsx
// app/dashboard/assistant/page.tsx
"use client";

import { MessageList } from "@/components/chat/message-list";
import { ChatInput } from "@/components/chat/chat-input";
import { ThreadSwitcher } from "@/components/chat/thread-switcher";

export default function AssistantPage() {
  return (
    <div className="flex h-[calc(100vh-64px)]">
      {/* Sidebar with threads */}
      <aside className="w-72 border-r border-slate-200 bg-slate-50 p-4 overflow-y-auto">
        <h2 className="mb-4 text-xs font-semibold uppercase tracking-wider text-slate-400">
          Conversations
        </h2>
        <ThreadSwitcher />
      </aside>

      {/* Main chat area */}
      <main className="flex flex-1 flex-col bg-white">
        <MessageList />
        <ChatInput />
      </main>
    </div>
  );
}
```

## Key Takeaways

- **Zero UI from the library** -- Every component above is custom. The library only provides state and actions through hooks.
- **Your design system** -- Colors, spacing, typography, and interactions are entirely yours.
- **Same backend** -- The headless hooks connect to the exact same Aomi backend as the widget. All tools, namespaces, and models are available.
- **Incremental adoption** -- You can start with the widget and replace components one at a time with custom versions, since both approaches use `@aomi-labs/react` under the hood.
